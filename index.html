PyLJFluid: Classical Fluid Simulations with Python

Reproduce and extend the original computer experiments with Lennard-Jones fluids using modern tools.

Get the source on github 
'''Simulate an Lennard-Jones fluid and plot g(r) at equilibrium
'''

import matplotlib.pyplot as plt
from pyljfluid import (LJForceField, Config, MDSimulator, 
                       PairCorrelationFunctionCalculator)

# Parameters
N = 1000
rho = 0.55
T = 1.5
m = 1.0
dt = 0.0025
forcefield = LJForceField(sigma=1.0, epsilon=1.0, r_cutoff=3.3)

# Initialze system
config0 = Config.create(N, rho, dt, forcefield.sigma, T, m)
sim = MDSimulator(config, forcefield)
sim.minimize_until(1.0)

# Equilibrate
for i in xrange(100):
    sim.rescale_velocities()
    sim.cycle(1000)

# Produce data
pcc = PairCorrelationFunctionCalculator(0.01, config0.box_size / 2)
for i in xrange(100):
    sim.cycle(1000)
    pcc.accumulate_config(sim.config)
r,g = pcc.calculate()

# Plot
plt.clf()
plt.plot(r, g)
plt.xlabel('Pair Separation, $r$ ($\sigma$)')
plt.ylabel('Pair Correlation, $g(r)$')
plt.show()
